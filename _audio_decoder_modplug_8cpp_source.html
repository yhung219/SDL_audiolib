33 static void
34 initModPlug(const SDL_AudioSpec& spec)
35 {
36  ModPlug_GetSettings(&modplugSettings);
37  modplugSettings.mFlags = MODPLUG_ENABLE_OVERSAMPLING | MODPLUG_ENABLE_NOISE_REDUCTION;
38  // TODO: can modplug handle more than 2 channels?
39  modplugSettings.mChannels = spec.channels == 1 ? 1 : 2;
40  // It seems MogPlug does resample to any samplerate. 32, 44.1, up to
41  // 192K all seem to work correctly.
42  modplugSettings.mFrequency = spec.freq;
43  modplugSettings.mResamplingMode = MODPLUG_RESAMPLE_FIR;
44  modplugSettings.mBits = 32;
45  ModPlug_SetSettings(&modplugSettings);
46  initialized = true;
47 }

82 bool
83 Aulib::AudioDecoderModPlug::open(SDL_RWops* rwops)
84 {
85  if (isOpen()) {
86  return true;
87  }
88  //FIXME: error reporting
89  Sint64 dataSize = SDL_RWsize(rwops);
90  if (dataSize <= 0 or dataSize > std::numeric_limits<int>::max()) {
91  return false;
92  }
93  Buffer<Uint8> data(dataSize);
94  if (SDL_RWread(rwops, data.get(), data.size(), 1) != 1) {
95  return false;
96  }
97  d->mpHandle.reset(ModPlug_Load(data.get(), (int)data.size()));
98  if (not d->mpHandle) {
99  return false;
100  }
101  ModPlug_SetMasterVolume(d->mpHandle.get(), 192);
102  d->fDuration = (float)ModPlug_GetLength(d->mpHandle.get()) / 1000.f;
103  setIsOpen(true);
104  return true;
105 }

122 int
123 Aulib::AudioDecoderModPlug::doDecoding(float buf[], int len, bool& callAgain)
124 {
125  callAgain = false;
126  if (d->atEOF) {
127  return 0;
128  }
129  Buffer<Sint32> tmpBuf(len);
130  int ret = ModPlug_Read(d->mpHandle.get(), tmpBuf.get(), len * 4);
131  // Convert from 32-bit to float.
132  for (int i = 0; i < len; ++i) {
133  buf[i] = (float)tmpBuf[i] / 2147483648.f;
134  }
135  if (ret == 0) {
136  d->atEOF = true;
137  }
138  return ret / sizeof(*buf);
139 }
