138 bool
139 Aulib::AudioDecoderMpg123::open(SDL_RWops* rwops)
140 {
141  if (isOpen()) {
142  return true;
143  }
144  if (not initialized) {
145  return false;
146  }
147  if ((d->fMpgHandle = mpg123_new(nullptr, nullptr)) == nullptr) {
148  return false;
149  }
150  if (initMpgFormats(d->fMpgHandle) < 0) {
151  mpg123_delete(d->fMpgHandle);
152  d->fMpgHandle = nullptr;
153  return false;
154  }
155  mpg123_replace_reader_handle(d->fMpgHandle, mpgReadCallback, mpgSeekCallback, nullptr);
156  mpg123_open_handle(d->fMpgHandle, rwops);
157  long rate;
158  int channels, encoding;
159  if (mpg123_getformat(d->fMpgHandle, &rate, &channels, &encoding) != 0) {
160  mpg123_delete(d->fMpgHandle);
161  d->fMpgHandle = nullptr;
162  return false;
163  }
164  d->fChannels = channels;
165  d->fRate = rate;
166  off_t len = mpg123_length(d->fMpgHandle);
167  d->fDuration = (len == MPG123_ERR) ? -1 : ((float)len / rate);
168  setIsOpen(true);
169  return true;
170 }

187 size_t
188 Aulib::AudioDecoderMpg123::doDecoding(float buf[], size_t len, bool& callAgain)
189 {
190  callAgain = false;
191  if (d->fEOF) {
192  return 0;
193  }
194 
195  size_t bytesWanted = len * sizeof(*buf);
196  size_t decBytes = 0;
197  size_t totalBytes = 0;
198 
199  while (totalBytes < bytesWanted and not callAgain) {
200  int ret = mpg123_read(d->fMpgHandle, (unsigned char*)buf, bytesWanted, &decBytes);
201  totalBytes += decBytes;
202  if (ret == MPG123_NEW_FORMAT) {
203  long rate;
204  int channels, encoding;
205  mpg123_getformat(d->fMpgHandle, &rate, &channels, &encoding);
206  d->fChannels = channels;
207  d->fRate = rate;
208  callAgain = true;
209  } else if (ret == MPG123_DONE) {
210  d->fEOF = true;
211  break;
212  }
213  }
214  return totalBytes / sizeof(*buf);
215 }
