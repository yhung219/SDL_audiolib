43 static int
44 initMpgFormats(mpg123_handle* handle)
45 {
46  const long* list;
47  size_t len;
48  mpg123_rates(&list, &len);
49  mpg123_format_none(handle);
50  for (size_t i = 0; i < len; ++i) {
51  if (mpg123_format(handle, list[i], MPG123_STEREO | MPG123_MONO, MPG123_ENC_FLOAT_32)

55  }
56  }

122 bool
123 Aulib::AudioDecoderMpg123::open(SDL_RWops* rwops)
124 {
125  if (isOpen()) {
126  return true;
127  }
128  if (not initialized) {
129  return false;
130  }
131  d->fMpgHandle.reset(mpg123_new(nullptr, nullptr));
132  if (not d->fMpgHandle) {
133  return false;
134  }
135  if (initMpgFormats(d->fMpgHandle.get()) < 0) {
136  return false;
137  }
138  mpg123_replace_reader_handle(d->fMpgHandle.get(), mpgReadCallback, mpgSeekCallback, nullptr);
139  mpg123_open_handle(d->fMpgHandle.get(), rwops);
140  long rate;
141  int channels, encoding;
142  if (mpg123_getformat(d->fMpgHandle.get(), &rate, &channels, &encoding) != 0) {
143  return false;
144  }
145  d->fChannels = channels;
146  d->fRate = rate;
147  off_t len = mpg123_length(d->fMpgHandle.get());
148  d->fDuration = (len == MPG123_ERR) ? -1 : ((float)len / rate);
149  setIsOpen(true);
150  return true;
151 }

168 int
169 Aulib::AudioDecoderMpg123::doDecoding(float buf[], int len, bool& callAgain)
170 {
171  callAgain = false;
172  if (d->fEOF) {
173  return 0;
174  }
175 
176  int bytesWanted = len * sizeof(*buf);
177  size_t decBytes = 0;
178  int totalBytes = 0;
179 
180  while (totalBytes < bytesWanted and not callAgain) {
181  int ret = mpg123_read(d->fMpgHandle.get(), (unsigned char*)buf, bytesWanted, &decBytes);
182  totalBytes += decBytes;
183  if (ret == MPG123_NEW_FORMAT) {
184  long rate;
185  int channels, encoding;
186  mpg123_getformat(d->fMpgHandle.get(), &rate, &channels, &encoding);
187  d->fChannels = channels;
188  d->fRate = rate;
189  callAgain = true;
190  } else if (ret == MPG123_DONE) {
191  d->fEOF = true;
192  break;
193  }
194  }
195  return totalBytes / sizeof(*buf);
196 }
