97 bool
98 Aulib::AudioDecoderSndfile::open(SDL_RWops* rwops)
99 {
100  if (isOpen()) {
101  return true;
102  }
103  d->fInfo.format = 0;
104  SF_VIRTUAL_IO cbs;
105  cbs.get_filelen = sfLenCb;
106  cbs.seek = sfSeekCb;
107  cbs.read = sfReadCb;
108  cbs.write = nullptr;
109  cbs.tell = sfTellCb;
110  d->fSndfile.reset(sf_open_virtual(&cbs, SFM_READ, &d->fInfo, rwops));
111  if (not d->fSndfile) {
112  return false;
113  }
114  d->fDuration = (float)d->fInfo.frames / d->fInfo.samplerate;
115  setIsOpen(true);
116  return true;
117 }
