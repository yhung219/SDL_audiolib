16 /* Relocate any samples in the specified buffer to the beginning:
17  *
18  * ....ssss -> ssss....
19  *
20  * The tracking indices are adjusted as needed.
21  */
22 static void
23 relocateBuffer(float* buf, int& pos, int& end)
24 {
25  if (end < 1) {
26  return;
27  }
28  if (pos >= end) {
29  pos = end = 0;
30  return;
31  }
32  if (pos < 1) {
33  return;
34  }
35  int len = end - pos;
36  memmove(buf, buf + pos, static_cast<size_t>(len) * sizeof(*buf));
37  pos = 0;
38  end = len;
39 }

88 int
89 Aulib::AudioResampler_priv::fMoveFromOutBuffer(float dst[], int dstLen)
90 {
91  if (fOutBufferEnd == 0) {
92  return 0;
93  }
94  if (fOutBufferPos >= fOutBufferEnd) {
95  fOutBufferPos = fOutBufferEnd = 0;
96  return 0;
97  }
98  int len = std::min(fOutBufferEnd - fOutBufferPos, dstLen);
99  memcpy(dst, fOutBuffer.get() + fOutBufferPos,
100  static_cast<size_t>(len) * sizeof(*fOutBuffer.get()));
101  fOutBufferPos += len;
102  if (fOutBufferPos >= fOutBufferEnd) {
103  fOutBufferEnd = fOutBufferPos = 0;
104  }
105  return len;
106 }

109 void
110 Aulib::AudioResampler_priv::fAdjustBufferSizes()
111 {
112  int oldInBufLen = fInBufferEnd - fInBufferPos;
113  int outBufSiz = fChannels * fChunkSize;
114  int inBufSiz;
115 
116  if (fDstRate == fSrcRate) {
117  // In the no-op case where we don't actually resample, input and output
118  // buffers have the same size, since we're just copying the samples
119  // as-is from input to output.
120  inBufSiz = outBufSiz;
121  } else {
122  // When resampling, the input buffer's size depends on the ratio between
123  // the source and destination sample rates.
124  inBufSiz = std::ceil((double)outBufSiz * fSrcRate / fDstRate);
125  auto remainder = inBufSiz % fChannels;
126  if (remainder != 0) {
127  inBufSiz = inBufSiz + fChannels - remainder;
128  }
129  }
130 
131  fOutBuffer.reset(outBufSiz);
132  fInBuffer.resize(inBufSiz);
133  fOutBufferPos = fOutBufferEnd = fInBufferPos = 0;
134  if (oldInBufLen != 0) {
135  fInBufferEnd = oldInBufLen;
136  } else {
137  fInBufferEnd = 0;
138  }
139 }

142 void
143 Aulib::AudioResampler_priv::fResampleFromInBuffer()
144 {
145  int inLen = fInBufferEnd - fInBufferPos;
146  float* from = fInBuffer.get() + fInBufferPos;
147  float* to = fOutBuffer.get() + fOutBufferEnd;
148  if (fSrcRate == fDstRate) {
149  // No resampling is needed. Just copy the samples as-is.
150  int outLen = std::min(fOutBuffer.size() - fOutBufferEnd, inLen);
151  std::memcpy(to, from, static_cast<size_t>(outLen) * sizeof(*from));
152  fOutBufferEnd += outLen;
153  fInBufferPos += outLen;
154  } else {
155  int outLen = fOutBuffer.size() - fOutBufferEnd;
156  q->doResampling(to, from, outLen, inLen);
157  fOutBufferEnd += outLen;
158  fInBufferPos += inLen;
159  }
160  if (fInBufferPos >= fInBufferEnd) {
161  // No more samples left to resample. Mark the input buffer as empty.
162  fInBufferPos = fInBufferEnd = 0;
163  }
164 }

219 int
220 Aulib::AudioResampler::resample(float dst[], int dstLen)
221 {
222  int totalSamples = 0;
223  bool decEOF = false;
224 
225  if (d->fPendingSpecChange) {
226  // There's a spec change pending. Process any data that is still in our
227  // buffers using the current spec.
228  totalSamples += d->fMoveFromOutBuffer(dst, dstLen);
229  relocateBuffer(d->fOutBuffer.get(), d->fOutBufferPos, d->fOutBufferEnd);
230  d->fResampleFromInBuffer();
231  if (totalSamples >= dstLen) {
232  // There's still samples left in the output buffer, so don't change
233  // the spec yet.
234  return dstLen;
235  }
236  // Our buffers are empty, so we can change to the new spec.
237  setSpec(d->fDstRate, d->fChannels, d->fChunkSize);
238  d->fPendingSpecChange = false;
239  }
240 
241  // Keep resampling until we either produce the requested amount of output
242  // samples, or the decoder has no more samples to give us.
243  while (totalSamples < dstLen and not decEOF) {
244  // If the input buffer is not filled, get some more samples from the
245  // decoder.
246  if (d->fInBufferEnd < d->fInBuffer.size()) {
247  bool callAgain = false;
248  int decSamples = d->fDecoder->decode(d->fInBuffer.get() + d->fInBufferEnd,
249  d->fInBuffer.size() - d->fInBufferEnd,
250  callAgain);
251  // If the decoder indicated a spec change, process any data that is
252  // still in our buffers using the current spec.
253  if (callAgain) {
254  d->fInBufferEnd += decSamples;
255  d->fResampleFromInBuffer();
256  totalSamples += d->fMoveFromOutBuffer(dst + totalSamples, dstLen - totalSamples);
257  if (totalSamples >= dstLen) {
258  // There's still samples left in the output buffer. Keep
259  // the current spec and prepare to change it on our next
260  // call.
261  d->fPendingSpecChange = true;
262  return dstLen;
263  }
264  setSpec(d->fDstRate, d->fChannels, d->fChunkSize);
265  } else if (decSamples <= 0) {
266  decEOF = true;
267  } else {
268  d->fInBufferEnd += decSamples;
269  }
270  }
271 
272  d->fResampleFromInBuffer();
273  relocateBuffer(d->fInBuffer.get(), d->fInBufferPos, d->fInBufferEnd);
274  totalSamples += d->fMoveFromOutBuffer(dst + totalSamples, dstLen - totalSamples);
275  relocateBuffer(d->fOutBuffer.get(), d->fOutBufferPos, d->fOutBufferEnd);
276  }
277  return totalSamples;
278 }
