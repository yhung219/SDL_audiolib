106 bool
107 Aulib::AudioDecoderFluidSynth::open(SDL_RWops* rwops)
108 {
109  if (isOpen()) {
110  return true;
111  }
112  if (d->fSynth == nullptr) {
113  return false;
114  }
115 
116  //FIXME: error reporting
117  Sint64 midiDataLen = SDL_RWsize(rwops);
118  if (midiDataLen <= 0) {
119  return false;
120  }
121  Buffer<Uint8> newMidiData((size_t)midiDataLen);
122  if (SDL_RWread(rwops, newMidiData.get(), newMidiData.size(), 1) != 1
123  or (d->fPlayer = new_fluid_player(d->fSynth)) == nullptr
124  or fluid_player_add_mem(d->fPlayer, newMidiData.get(), newMidiData.size()) != FLUID_OK
125  or fluid_player_play(d->fPlayer) != FLUID_OK)
126  {
127  if (d->fPlayer) {
128  delete_fluid_player(d->fPlayer);
129  }
130  if (d->fSynth) {
131  delete_fluid_synth(d->fSynth);
132  d->fSynth = nullptr;
133  }
134  return false;
135  }
136  d->fMidiData.swap(newMidiData);
137  setIsOpen(true);
138  return true;
139 }

160 size_t
161 Aulib::AudioDecoderFluidSynth::doDecoding(float buf[], size_t len, bool& callAgain)
162 {
163  callAgain = false;
164  if (d->fSynth == nullptr or d->fEOF) {
165  return 0;
166  }
167 
168  len /= Aulib::spec().channels;
169  int res = fluid_synth_write_float(d->fSynth, len, buf, 0, 2, buf, 1, 2);
170  if (fluid_player_get_status(d->fPlayer) == FLUID_PLAYER_DONE) {
171  d->fEOF = true;
172  }
173  if (res == FLUID_OK) {
174  return len * Aulib::spec().channels;
175  }
176  return 0;
177 }
