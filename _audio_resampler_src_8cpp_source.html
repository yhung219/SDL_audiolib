44 void
45 Aulib::AudioResamplerSRC::doResampling(float dst[], const float src[], int& dstLen, int& srcLen)
46 {
47  if (not d->fResampler) {
48  dstLen = srcLen = 0;
49  return;
50  }
51 
52  d->fData.data_in = const_cast<float*>(src);
53  d->fData.data_out = dst;
54  int channels = currentChannels();
55  d->fData.input_frames = srcLen / channels;
56  d->fData.output_frames = dstLen / channels;
57  d->fData.end_of_input = false;
58 
59  src_process(d->fResampler.get(), &d->fData);
60 
61  dstLen = d->fData.output_frames_gen * channels;
62  srcLen = d->fData.input_frames_used * channels;
63 }
