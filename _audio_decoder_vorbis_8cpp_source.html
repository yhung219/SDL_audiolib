103 bool
104 Aulib::AudioDecoderVorbis::open(SDL_RWops* rwops)
105 {
106  if (isOpen()) {
107  return true;
108  }
109  ov_callbacks cbs;
110  memset(&cbs, 0, sizeof(cbs));
111  cbs.read_func = vorbisReadCb;
112  cbs.seek_func = vorbisSeekCb;
113  cbs.tell_func = vorbisTellCb;
114  cbs.close_func = nullptr;
115  d->fVFHandle = new OggVorbis_File;
116  if (ov_open_callbacks(rwops, d->fVFHandle, nullptr, 0, cbs) != 0) {
117  delete d->fVFHandle;
118  d->fVFHandle = nullptr;
119  return false;
120  }
121  d->fCurrentInfo = ov_info(d->fVFHandle, -1);
122  double len = ov_time_total(d->fVFHandle, -1);
123  d->fDuration = len == OV_EINVAL ? -1 : len;
124  setIsOpen(true);
125  return true;
126 }

143 int
144 Aulib::AudioDecoderVorbis::doDecoding(float buf[], int len, bool& callAgain)
145 {
146  callAgain = false;
147 
148  if (d->fEOF) {
149  return 0;
150  }
151 
152  float** out;
153  int decSamples = 0;
154 
155  while (decSamples < len and not callAgain) {
156  int lastSection = d->fCurrentSection;
157  //TODO: We only support up to 2 channels for now.
158  int channels = std::min(d->fCurrentInfo->channels, 2);
159  long ret = ov_read_float(d->fVFHandle, &out, (len - decSamples) / channels, &d->fCurrentSection);
160  if (ret == 0) {
161  d->fEOF = true;
162  break;
163  }
164  if (ret < 0) {
165  AM_debugPrint("libvorbis stream error: ");
166  switch (ret) {
167  case OV_HOLE: AM_debugPrintLn("OV_HOLE"); break;
168  case OV_EBADLINK: AM_debugPrintLn("OV_EBADLINK"); break;
169  case OV_EINVAL: AM_debugPrintLn("OV_EINVAL"); break;
170  default: AM_debugPrintLn("unknown error: " << ret);
171  }
172  break;
173  }
174  if (d->fCurrentSection != lastSection) {
175  d->fCurrentInfo = ov_info(d->fVFHandle, -1);
176  callAgain = true;
177  }
178  // Copy samples to output buffer in interleaved format.
179  for (int samp = 0; samp < ret; ++samp) {
180  for (int chan = 0; chan < channels; ++chan) {
181  *buf++ = out[chan][samp];
182  }
183  }
184  decSamples += ret * channels;
185  }
186  return decSamples;
187 }
