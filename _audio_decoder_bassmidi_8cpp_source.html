101 bool
102 Aulib::AudioDecoderBassmidi::open(SDL_RWops* rwops)
103 {
104  if (isOpen()) {
105  return true;
106  }
107 
108  int frontPos = SDL_RWtell(rwops);
109  d->midiDataLen = SDL_RWseek(rwops, 0, RW_SEEK_END) - frontPos;
110  if (d->midiDataLen <= 0) {
111  return false;
112  }
113  SDL_RWseek(rwops, frontPos, RW_SEEK_SET);
114  d->midiData = new Uint8[d->midiDataLen];
115  DWORD bassFlags = BASS_SAMPLE_FLOAT | BASS_STREAM_DECODE | BASS_MIDI_DECAYEND | BASS_MIDI_SINCINTER;
116  if (SDL_RWread(rwops, d->midiData, d->midiDataLen, 1) != 1
117  or (d->hstream = BASS_MIDI_StreamCreateFile(TRUE, d->midiData, 0, d->midiDataLen, bassFlags, 1)) == 0)
118  {
119  if (d->hstream) {
120  BASS_StreamFree(d->hstream);
121  d->hstream = 0;
122  } else {
123  AM_debugPrintLn("AudioDecoderBassmidi: got BASS error " << BASS_ErrorGetCode()
124  << " while creating HSTREAM.");
125  }
126  delete[] d->midiData;
127  d->midiData = nullptr;
128  return false;
129  }
130  setIsOpen(true);
131  return true;
132 }

155 int
156 Aulib::AudioDecoderBassmidi::doDecoding(float buf[], int len, bool&)
157 {
158  if (d->eof or d->hstream == 0) {
159  return 0;
160  }
161 
162  DWORD byteLen = len * sizeof(*buf);
163  DWORD ret = BASS_ChannelGetData(d->hstream, buf, byteLen | BASS_DATA_FLOAT);
164  if (ret == (DWORD)-1) {
165  SDL_SetError("AudioDecoderBassmidi: got BASS error %d during decoding.\n", BASS_ErrorGetCode());
166  return 0;
167  } else if (ret < byteLen) {
168  d->eof = true;
169  }
170  return ret / sizeof(*buf);
171 }
