54 void
55 Aulib::AudioResamplerSpeex::doResampling(float dst[], const float src[], int& dstLen,
56  int& srcLen)
57 {
58  if (not d->fResampler) {
59  dstLen = srcLen = 0;
60  return;
61  }
62 
63  int channels = currentChannels();
64  auto spxInLen = static_cast<spx_uint32_t>(srcLen / channels);
65  auto spxOutLen = static_cast<spx_uint32_t>(dstLen / channels);
66  if (spxInLen == 0 or spxOutLen == 0) {
67  dstLen = srcLen = 0;
68  return;
69  }
70  speex_resampler_process_interleaved_float(d->fResampler.get(), src, &spxInLen, dst, &spxOutLen);
71  dstLen = static_cast<int>(spxOutLen) * channels;
72  srcLen = static_cast<int>(spxInLen) * channels;
73 }
