171 size_t
172 Aulib::AudioDecoderMusepack::doDecoding(float buf[], size_t len, bool& callAgain)
173 {
174  callAgain = false;
175  unsigned totalSamples = 0;
176  unsigned wantedSamples = len;
177 
178  if (d->eof) {
179  return 0;
180  }
181 
182  // If we have any left-over samples from the previous frame, copy them out.
183  if (d->curFrame.samples > 0) {
184  unsigned copyLen = std::min((size_t)d->curFrame.samples * d->strmInfo.channels, len);
185  std::memcpy(buf, d->curFrame.buffer + d->frameBufPos, copyLen * sizeof(*buf));
186  d->curFrame.samples -= copyLen / d->strmInfo.channels;
187  if (d->curFrame.samples != 0) {
188  // There's still more samples left.
189  d->frameBufPos += copyLen;
190  return copyLen;
191  }
192  buf += copyLen;
193  len -= copyLen;
194  totalSamples += copyLen;
195  d->frameBufPos = 0;
196  }
197 
198  // Decode one frame at a time, until we have enough samples.
199  while (totalSamples < wantedSamples) {
200  if (mpc_demux_decode(d->demuxer, &d->curFrame) != MPC_STATUS_OK) {
201  AM_warnLn("AudioDecoderMusepack decoding error.");
202  return 0;
203  }
204  unsigned copyLen = std::min((size_t)d->curFrame.samples * d->strmInfo.channels, len);
205  std::memcpy(buf, d->curFrame.buffer, copyLen * sizeof(*buf));
206  d->frameBufPos = copyLen;
207  d->curFrame.samples -= copyLen / d->strmInfo.channels;
208  totalSamples += copyLen;
209  len -= copyLen;
210  buf += copyLen;
211  if (d->curFrame.bits == -1) {
212  d->eof = true;
213  return totalSamples;
214  }
215  }
216  return totalSamples;
217 }
