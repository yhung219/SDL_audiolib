112 int
113 Aulib::AudioDecoderWildmidi::doDecoding(float buf[], int len, bool& callAgain)
114 {
115  callAgain = false;
116  if (not d->midiHandle or d->eof) {
117  return 0;
118  }
119 
120  if (d->sampBuf.size() != len) {
121  d->sampBuf.reset(len);
122  }
123  int res = WildMidi_GetOutput(d->midiHandle.get(), (char*)d->sampBuf.get(),
124  static_cast<unsigned long>(len) * 2);
125  if (res < 0) {
126  return 0;
127  }
128  // Convert from 16-bit to float.
129  for (int i = 0; i < res / 2; ++i) {
130  buf[i] = (float)d->sampBuf[i] / 32768.f;
131  }
132  if (res < len) {
133  d->eof = true;
134  }
135  return res / 2;
136 }
