49 bool
50 Aulib::AudioDecoderXmp::open(SDL_RWops* rwops)
51 {
52  if (isOpen()) {
53  return true;
54  }
55 
56  //FIXME: error reporting
57  d->fContext.reset(xmp_create_context());
58  if (not d->fContext) {
59  return false;
60  }
61 
62  Sint64 dataSize = SDL_RWsize(rwops);
63  if (dataSize <= 0 or dataSize > std::numeric_limits<int>::max()) {
64  return false;
65  }
66  Buffer<Uint8> data(dataSize);
67  if (SDL_RWread(rwops, data.get(), data.size(), 1) != 1) {
68  return false;
69  }
70  if (xmp_load_module_from_memory(d->fContext.get(), data.get(), data.size()) != 0) {
71  return false;
72  }
73  // libXMP supports 8-48kHz.
74  d->fRate = std::min(std::max(8000, Aulib::spec().freq), 48000);
75  if (xmp_start_player(d->fContext.get(), d->fRate, 0) != 0) {
76  return false;
77  }
78  setIsOpen(true);
79  return true;
80 }

119 int
120 Aulib::AudioDecoderXmp::doDecoding(float buf[], int len, bool& callAgain)
121 {
122  callAgain = false;
123  if (d->fEof) {
124  return 0;
125  }
126  Buffer<Sint16> tmpBuf(len);
127  auto ret = xmp_play_buffer(d->fContext.get(), tmpBuf.get(), len * 2, 1);
128  // Convert from 16-bit to float.
129  for (int i = 0; i < len; ++i) {
130  buf[i] = (float)tmpBuf[i] / 32768.f;
131  }
132  if (ret == -XMP_END) {
133  d->fEof = true;
134  }
135  if (ret < 0) {
136  return 0;
137  }
138  return len;
139 }
