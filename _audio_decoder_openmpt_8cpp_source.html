48 bool
49 Aulib::AudioDecoderOpenmpt::open(SDL_RWops* rwops)
50 {
51  if (isOpen()) {
52  return true;
53  }
54  //FIXME: error reporting
55  Sint64 dataSize = SDL_RWsize(rwops);
56  if (dataSize <= 0 or dataSize > std::numeric_limits<int>::max()) {
57  return false;
58  }
59  Buffer<Uint8> data(dataSize);
60  if (SDL_RWread(rwops, data.get(), data.size(), 1) != 1) {
61  return false;
62  }
63 
64  std::unique_ptr<openmpt::module> module(nullptr);
65  try {
66  module = std::make_unique<openmpt::module>(data.get(), data.size());
67  } catch (const openmpt::exception& e) {
68  AM_warnLn("libopenmpt failed to load mod: " << e.what());
69  return false;
70  }
71 
72  d->fDuration = module->get_duration_seconds();
73  d->fModule.swap(module);
74  setIsOpen(true);
75  return true;
76 }

117 int
118 Aulib::AudioDecoderOpenmpt::doDecoding(float buf[], int len, bool& callAgain)
119 {
120  callAgain = false;
121  if (d->atEOF) {
122  return 0;
123  }
124  int ret;
125  if (Aulib::spec().channels == 2) {
126  ret = d->fModule->read_interleaved_stereo(Aulib::spec().freq, len / 2, buf) * 2;
127  } else {
128  AM_debugAssert(Aulib::spec().channels == 1);
129  ret = d->fModule->read(Aulib::spec().freq, len, buf);
130  }
131  if (ret == 0) {
132  d->atEOF = true;
133  }
134  return ret;
135 }
