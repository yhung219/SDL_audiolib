175 // Conversion happens in-place.
176 static void
177 monoToStereo(float buf[], int len)
178 {
179  if (len < 1 or buf == nullptr) {
180  return;
181  }
182  for (int i = len / 2 - 1, j = len - 1; i > 0; --i) {
183  buf[j--] = buf[i];
184  buf[j--] = buf[i];
185  }
186 }

189 static void
190 stereoToMono(float dst[], float src[], int srcLen)
191 {
192  if (srcLen < 1 or dst == nullptr or src == nullptr) {
193  return;
194  }
195  for (int i = 0, j = 0; i < srcLen; i += 2, ++j) {
196  dst[j] = src[i] * 0.5f;
197  dst[j] += src[i + 1] * 0.5f;
198  }
199 }

202 int
203 Aulib::AudioDecoder::decode(float buf[], int len, bool& callAgain)
204 {
205  const SDL_AudioSpec& spec = Aulib::spec();
206 
207  if (this->getChannels() == 1 and spec.channels == 2) {
208  int srcLen = this->doDecoding(buf, len / 2, callAgain);
209  monoToStereo(buf, srcLen * 2);
210  return srcLen * 2;
211  }
212 
213  if (this->getChannels() == 2 and spec.channels == 1) {
214  if (d->stereoBuf.size() != len * 2) {
215  d->stereoBuf.reset(len * 2);
216  }
217  int srcLen = this->doDecoding(d->stereoBuf.get(), d->stereoBuf.size(), callAgain);
218  stereoToMono(buf, d->stereoBuf.get(), srcLen);
219  return srcLen / 2;
220  }
221  return this->doDecoding(buf, len, callAgain);
222 }
