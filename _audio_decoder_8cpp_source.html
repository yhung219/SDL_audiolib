124 static void
125 monoToStereo(float buf[], size_t len)
126 {
127  for (size_t i = len / 2 - 1, j = len - 1; i > 0; --i) {
128  buf[j--] = buf[i];
129  buf[j--] = buf[i];

134 static void
135 stereoToMono(float dst[], float src[], size_t srcLen)
136 {
137  for (size_t i = 0, j = 0; i < srcLen; i += 2, ++j) {
138  dst[j] = src[i] * 0.5f;
139  dst[j] += src[i + 1] * 0.5f;
140  }
141 }

144 size_t
145 Aulib::AudioDecoder::decode(float buf[], size_t len, bool& callAgain)
146 {
147  const SDL_AudioSpec& spec = Aulib::spec();
148 
149  if (this->getChannels() == 1 and spec.channels == 2) {
150  size_t srcLen = this->doDecoding(buf, len / 2, callAgain);
151  monoToStereo(buf, srcLen * 2);
152  return srcLen * 2;
153  }
154 
155  if (this->getChannels() == 2 and spec.channels == 1) {
156  if (d->stereoBuf.size() != len * 2) {
157  d->stereoBuf.reset(len * 2);
158  }
159  size_t srcLen = this->doDecoding(d->stereoBuf.get(), d->stereoBuf.size(), callAgain);
160  stereoToMono(buf, d->stereoBuf.get(), srcLen);
161  return srcLen / 2;
162  }
163  return this->doDecoding(buf, len, callAgain);
164 }
