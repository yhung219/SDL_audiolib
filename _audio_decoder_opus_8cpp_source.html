92 bool
93 Aulib::AudioDecoderOpus::open(SDL_RWops* rwops)
94 {
95  // FIXME: Check if already opened.
96  static OpusFileCallbacks cbs;
97  memset(&cbs, 0, sizeof(cbs));
98  cbs.read = opusReadCb;
99  cbs.seek = opusSeekCb;
100  cbs.tell = opusTellCb;
101  cbs.close = nullptr;
102  if (d->fOpusHandle) {
103  op_free(d->fOpusHandle);
104  d->fOpusHandle = nullptr;
105  }
106  int error;
107  if ((d->fOpusHandle = op_open_callbacks(rwops, &cbs, nullptr, 0, &error)) == nullptr) {
108  AM_debugPrintLn("ERROR:" << error);
109  if (error == OP_ENOTFORMAT) {
110  AM_debugPrintLn("OP_ENOTFORMAT");
111  }
112  return false;
113  }
114  ogg_int64_t len = op_pcm_total(d->fOpusHandle, -1);
115  // Opus is always 48kHz.
116  d->fDuration = len == OP_EINVAL ? -1 : (float)len / 48000.f;
117  return true;
118 }

135 int
136 Aulib::AudioDecoderOpus::doDecoding(float buf[], int len, bool& callAgain)
137 {
138  callAgain = false;
139 
140  if (d->fEOF) {
141  return 0;
142  }
143 
144  int decSamples = 0;
145 
146  while (decSamples < len) {
147  int ret = op_read_float_stereo(d->fOpusHandle, buf + decSamples, len - decSamples);
148  if (ret == 0) {
149  d->fEOF = true;
150  break;
151  }
152  if (ret < 0) {
153  AM_debugPrint("libvorbis stream error: ");
154  switch (ret) {
155  case OP_HOLE: AM_debugPrintLn("OV_HOLE"); break;
156  case OP_EBADLINK: AM_debugPrintLn("OV_EBADLINK"); break;
157  case OP_EINVAL: AM_debugPrintLn("OV_EINVAL"); break;
158  default: AM_debugPrintLn("unknown error: " << ret);
159  }
160  break;
161  }
162  decSamples += ret * 2;
163  }
164  return decSamples;
165 }
