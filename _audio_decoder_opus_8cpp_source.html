59 bool
60 Aulib::AudioDecoderOpus::open(SDL_RWops* rwops)
61 {
62  if (isOpen()) {
63  return true;
64  }
65  int error;
66  d->fOpusHandle.reset(op_open_callbacks(rwops, &d->fCbs, nullptr, 0, &error));
67  if (not d->fOpusHandle) {
68  AM_debugPrintLn("ERROR:" << error);
69  if (error == OP_ENOTFORMAT) {
70  AM_debugPrintLn("OP_ENOTFORMAT");
71  }
72  return false;
73  }
74  ogg_int64_t len = op_pcm_total(d->fOpusHandle.get(), -1);
75  // Opus is always 48kHz.
76  d->fDuration = len == OP_EINVAL ? -1 : (float)len / 48000.f;
77  setIsOpen(true);
78  return true;
79 }

96 int
97 Aulib::AudioDecoderOpus::doDecoding(float buf[], int len, bool& callAgain)
98 {
99  callAgain = false;
100 
101  if (d->fEOF) {
102  return 0;
103  }
104 
105  int decSamples = 0;
106 
107  while (decSamples < len) {
108  int ret = op_read_float_stereo(d->fOpusHandle.get(), buf + decSamples, len - decSamples);
109  if (ret == 0) {
110  d->fEOF = true;
111  break;
112  }
113  if (ret < 0) {
114  AM_debugPrint("libopusfile stream error: ");
115  switch (ret) {
116  case OP_HOLE: AM_debugPrintLn("OP_HOLE"); break;
117  case OP_EBADLINK: AM_debugPrintLn("OP_EBADLINK"); break;
118  case OP_EINVAL: AM_debugPrintLn("OP_EINVAL"); break;
119  default: AM_debugPrintLn("unknown error: " << ret);
120  }
121  break;
122  }
123  decSamples += ret * 2;
124  }
125  return decSamples;
126 }
