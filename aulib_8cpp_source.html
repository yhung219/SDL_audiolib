42 int
43 Aulib::init(int freq, int format, int channels, int bufferSize)
44 {
45  if (SDL_InitSubSystem(SDL_INIT_AUDIO) != 0) {
46  return -1;
47  }
48 
49  // We only support mono and stereo at this point.
50  channels = std::min(std::max(1, channels), 2);
51 
52  SDL_AudioSpec requestedSpec;
53  memset(&requestedSpec, 0, sizeof(::SDL_AudioSpec));
54  requestedSpec.freq = freq;
55  requestedSpec.format = format;
56  requestedSpec.channels = channels;
57  requestedSpec.samples = bufferSize;
58  requestedSpec.userdata = nullptr;
59  requestedSpec.callback = ::sdlCallback;
60  AudioStream_priv::fAudioSpec = requestedSpec;
61  if (SDL_OpenAudio(&requestedSpec, &AudioStream_priv::fAudioSpec) != 0) {
62  Aulib::quit();
63  return -1;
64  }
65 
66  AM_debugPrint("SDL initialized with sample format: ");
67  switch (AudioStream_priv::fAudioSpec.format) {
68  case AUDIO_S8:
69  AM_debugPrintLn("S8");
70  AudioStream_priv::fSampleConverter = Aulib::floatToS8;
71  break;
72  case AUDIO_U8:
73  AM_debugPrintLn("U8");
74  AudioStream_priv::fSampleConverter = Aulib::floatToU8;
75  break;
76  case AUDIO_S16LSB:
77  AM_debugPrintLn("S16LSB");
78  AudioStream_priv::fSampleConverter = Aulib::floatToS16LSB;
79  break;
80  case AUDIO_U16LSB:
81  AM_debugPrintLn("U16LSB");
82  AudioStream_priv::fSampleConverter = Aulib::floatToU16LSB;
83  break;
84  case AUDIO_S16MSB:
85  AM_debugPrintLn("S16MSB");
86  AudioStream_priv::fSampleConverter = Aulib::floatToS16MSB;
87  break;
88  case AUDIO_U16MSB:
89  AM_debugPrintLn("U16MSB");
90  AudioStream_priv::fSampleConverter = Aulib::floatToU16MSB;
91  break;
92 #if SDL_VERSION_ATLEAST(1,3,0)
93  case AUDIO_S32LSB:
94  AM_debugPrintLn("S32LSB");
95  AudioStream_priv::fSampleConverter = Aulib::floatToS32LSB;
96  break;
97  case AUDIO_F32LSB:
98  AM_debugPrintLn("F32LSB");
99  AudioStream_priv::fSampleConverter = Aulib::floatToFloat;
100  break;
101 #endif
102  default:
103  AM_warnLn("Unknown audio format spec: " << AudioStream_priv::fAudioSpec.format);
104  Aulib::quit();
105  return -1;
106  }
107 
108  SDL_PauseAudio(false);
109  gInitialized = true;
110  std::atexit(Aulib::quit);
111  return 0;
112 }

115 void
116 Aulib::quit()
117 {

119  return;
120  }

122  if (AudioStream_priv::fStreamList[i]->isPlaying()) {
123  AudioStream_priv::fStreamList[i]->stop();
124  }
125  }
126  SDL_QuitSubSystem(SDL_INIT_AUDIO);
127  AudioStream_priv::fSampleConverter = nullptr;
128  gInitialized = false;
